# Semi-Supervised Learning for Bad Exposures in Large Imaging Surveys

ML pipeline for classifying DECam exposures as good/bad using DINOv2 embeddings + kNN.

## Quick start
```bash
pip install -r requirements.txt
# Package imports need src/ on PYTHONPATH (set in pytest.ini for tests):
export PYTHONPATH=$PWD/src:$PYTHONPATH
python -m decam_qa.cli embed --config configs/embed.yaml --dset data.csv --dr dr10
python -m decam_qa.cli train --config configs/train.yaml
python -m decam_qa.cli classify --config configs/inference.yaml
python -m pytest tests/ -v
```

## NERSC/Perlmutter

## Reason classification scheme
15 bad-exposure categories stored as bitmask integers. See `src/decam_qa/info.py`
for the full scheme. `decode_reason(bitmask)` converts to human-readable strings.

### 2. Reason classification (bitmask decoding)

```python
from decam_qa import decode_reason, decode_vi_source, decode_ml_label, reason_num_dict

# Single reason
print(decode_reason(2**2))  # ["Clouds_transparency"]

# Multiple reasons from bitmask
mask = 2**1 | 2**6           # Saturated + Ghost_scatter
print(decode_reason(mask))   # ["Saturated", "Ghost_scatter"]

# Get numeric indices instead of strings
print(decode_reason(mask, return_num=True))  # [1, 6]

# Round-trip: encode → decode → re-encode
names = decode_reason(mask)
reencoded = sum(2 ** reason_num_dict[n] for n in names)
assert reencoded == mask

# Visual inspection sources (bit 0=Rongpu, bit 1=Alex)
print(decode_vi_source(1))   # ["Rongpu"]
print(decode_vi_source(3))   # ["Rongpu", "Alex"]

# ML label integers → reason strings (0 = good, >=1 = category index+1)
print(decode_ml_label([0, 2, 5]))  # ["good", "Saturated", "Nonoptimal_exp"]
```

### 3. CCD focal plane lookups

### 6. Generating DINOv2 embeddings

```python
from decam_qa import DECamImageDataset, create_model, generate_embeddings

# Load dataset and model
ds = DECamImageDataset("data.csv", "/path/to/images")
model = create_model(size="base", use_register=True)  # DINOv2 ViT-B/14

# Generate embeddings (to HDF5 in embeds_out/)
generate_embeddings(
    ds, model,
    device="cuda",           # or "cpu"
    output_dir="./output",
    batch_size=4,
    num_workers=4,
    gpu_idx=0,
    crop_size=(2352, 1176),  # height × width before resize
)
```

### 7. Training the kNN classifier

```python
from decam_qa import read_embeddings, build_pipeline, train, predict
import numpy as np

# Load training data
train_data, _, train_label = read_embeddings("train/embeds_out")
X = np.vstack(train_data)           # (n_samples, 768)
y = np.array(train_label, dtype=int)

# Build and train pipeline
pipe = build_pipeline()
fitted = train(
    pipe, X, y,
    search_params={                   # hyperparameter grid
        "classifier__n_neighbors": [1, 3, 5, 10],
        "classifier__p": [1, 2],
        "preprocessor__n_components": [10, 30, 50],
    },
    test_size=0.3,
    random_state=42,
    n_jobs=4,
)

# Predict on new data
test_embeds = np.load("test_embeds.npy")
labels, probs = predict(fitted, test_embeds)

# Serialize for later use
import joblib
joblib.dump(fitted, "trained_pipeline.pkl", compress=1)

# Load and reuse
loaded = joblib.load("trained_pipeline.pkl")
labels2, probs2 = predict(loaded, test_embeds)
assert np.array_equal(labels, labels2)
```

### 11. End-to-end script

```python
"""Full pipeline: embeddings → training → inference."""
from decam_qa import (
    DECamImageDataset, create_model, generate_embeddings,
    read_embeddings, build_pipeline, train, predict,
    load_config,
)
import numpy as np, joblib

# ---- Stage 1: Embeddings ----
cfg_embed = load_config("configs/embed.yaml", "embed")
ds = DECamImageDataset("data/train.csv", cfg_embed["image_roots"]["dr10"])
model = create_model("base")
generate_embeddings(ds, model, device="cuda", output_dir="./run01")

# ---- Stage 2: Training ----
cfg_train = load_config("configs/train.yaml", "train")
data, _, labels = read_embeddings("./run01/embeds_out")
X = np.vstack(data)
y = np.array(labels, dtype=int)
pipe = build_pipeline()
fitted = train(pipe, X, y, test_size=0.3, random_state=42)
joblib.dump(fitted, "trained_pipeline.pkl")

# ---- Stage 3: Inference ----
cfg_inf = load_config("configs/inference.yaml", "inference")
test_data, idx, _ = read_embeddings(cfg_inf["embeddings_dir"])
X_test = np.vstack(test_data)
labels, probs = predict(joblib.load("trained_pipeline.pkl"), X_test)
print(f"Predicted {len(labels)} exposures, "
      f"{np.sum(labels > 0)} flagged as bad "
      f"(threshold={cfg_inf['probability_threshold']})")
```
